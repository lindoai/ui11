<script>
function toggleSingleAccordion(itemId) {
  // Close all other items
  document.querySelectorAll('.accordion-content').forEach(content => {
    if (content.id !== itemId) {
      content.classList.add('hidden');
      const icon = document.getElementById(content.id + '-icon');
      if (icon) icon.classList.remove('rotate-180');
    }
  });
  
  // Toggle the clicked item
  const content = document.getElementById(itemId);
  const icon = document.getElementById(itemId + '-icon');
  content.classList.toggle('hidden');
  icon.classList.toggle('rotate-180');
}
</script>

Usage Guidelines

  • • Use accordions to organize and show/hide related content
  • • Keep accordion headers concise and descriptive
  • • Consider single vs multiple open based on content relationship
  • • Include visual indicators for expand/collapse state

Accessibility Features

  • • Keyboard navigation support with Enter and Space keys
  • • ARIA attributes for screen reader compatibility
  • • Focus management and clear visual indicators
  • • Smooth animations that respect reduced motion preferences
